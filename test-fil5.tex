\directlua{
  local widen_count = 0
  function test_widen(head, groupcode)
    local fullw = 360 * 65536
    local narrow_tw = 220 * 65536
    local narrow_max = narrow_tw + 15 * 65536
    local current = head
    while current do
      if current.id == 0 then
        local lw = current.width / 65536.0
        if lw <= narrow_max and lw > 0 then
          widen_count = widen_count + 1
          current.width = fullw
          local inner_head = current.head
          if inner_head then
            local inner_last = node.tail(inner_head)
            local fil = node.new(node.id("glue"))
            fil.stretch = 65536
            fil.stretch_order = 2
            if inner_last then
              node.insert_after(inner_head, inner_last, fil)
            end
          end
        end
      end
      current = current.next
    end
    return head
  end
  luatexbase.add_to_callback("post_linebreak_filter",
    function(head, groupcode)
      return test_widen(head, groupcode)
    end, "test_widen")
}

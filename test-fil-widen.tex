% Test: Does inserting fil glue inside an hbox at post_linebreak_filter
% actually stretch the text to full width?

\directlua{
  luatexbase.add_to_callback("post_linebreak_filter",
    function(head, groupcode)
      local current = head
      while current do
        if current.id == 0 then  -- hbox
          local w = current.width / 65536.0
          if w > 0 and w < 300 then  -- narrow line
            texio.write_nl("GHOST_WIDEN: found narrow line w=" .. tostring(w))
            current.width = 23592960  -- full width (~360pt)
            local inner_head = current.head
            if inner_head then
              local inner_last = node.tail(inner_head)
              local fil = node.new(node.id("glue"))
              fil.stretch = 65536
              fil.stretch_order = 2
              if inner_last then
                node.insert_after(inner_head, inner_last, fil)
                texio.write_nl("GHOST_WIDEN: inserted fil after node id=" .. tostring(inner_last.id))
              end
            end
          end
        end
        current = current.next
      end
      return head
    end, "test widen")
}

\begin{document}
% Create a narrow paragraph using parshape
\parshape 3 0pt 200pt 0pt 200pt 0pt 360pt
This is a narrow line of text that should be widened by the callback to full width.
This is the second narrow line that should also be widened.
This is the third line which is already full width and should NOT be widened.
\parshape 1 0pt 360pt
This paragraph is back to full width and should not be affected.
\end{document}

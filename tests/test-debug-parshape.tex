% Add extra diagnostics: track remaining after each paragraph
\directlua{
  local orig_post_lb = swarmwrap_post_lb
  function swarmwrap_post_lb_dbg(head, groupcode)
    local rem_before = tex.count["swarmwrap@remaining"]
    local tw_sp = tex.dimen["swarmwrap@tw@lua"]
    local tw_val = tw_sp / 65536.0
    head = orig_post_lb(head, groupcode)
    local rem_after = tex.count["swarmwrap@remaining"]
    local narrow_count = 0
    if tw_sp > 0 then
      local cur = head
      while cur do
        if cur.id == 0 then
          local lw = cur.width / 65536.0
          if lw <= tw_val + 3.0 and lw > 0 then
            narrow_count = narrow_count + 1
          end
        end
        cur = cur.next
      end
    end
    texio.write_nl("LOG: post_lb callback: rem_before=" .. rem_before 
      .. " rem_after=" .. rem_after 
      .. " narrow_count=" .. narrow_count
      .. " tw_sp=" .. string.format("%.1f", tw_val)
      .. " groupcode=" .. tostring(groupcode))
    return head
  end
  luatexbase.add_to_callback("post_linebreak_filter",
    swarmwrap_post_lb_dbg, "swarmwrap: debug post_lb")
}

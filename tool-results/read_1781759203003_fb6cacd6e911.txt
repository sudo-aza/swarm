     1→% swarmwrap.sty — Right-side float wrapper for the Swarm toolkit (v3.46)
     2→%
     3→% Text wrapping using \parshape applied by \swarmwrapnext command.
     4→% Built from scratch after all 9 existing LaTeX wrapping packages failed QA.
     5→%
     6→% Usage:
     7→%   \usepackage{swarmwrap}
     8→%
     9→%   % Simple figure (single content box):
    10→%   \begin{swarmwrap}%
    11→%     \includegraphics[width=3cm]{photo}%
    12→%   \end{swarmwrap}%
    13→%   \swarmwrapnext
    14→%   Paragraph text wraps around the figure.
    15→%
    16→%   % Multi-line content (figure + caption):
    17→%   \begin{swarmwrap}%
    18→%     \begin{minipage}[t]{3cm}
    19→%       \centering
    20→%       \includegraphics[width=\linewidth]{photo}
    21→%       \captionof{figure}{Caption.}
    22→%     \end{minipage}%
    23→%   \end{swarmwrap}%
    24→%   \swarmwrapnext
    25→%   Paragraph text wraps around the figure.
    26→%
    27→%   % Custom page-break penalty (discourage breaks in wrapped zone):
    28→%   \swarmwrappenalty{500}  % default is 10000 (very discouraged)
    29→%
    30→%   % Custom text-figure gap (horizontal space between text and figure):
    31→%   \swarmwrapgap{20pt}  % default is 14pt
    32→%
    33→%   % Custom squeeze-fit threshold (fraction of figure height):
    34→%   \swarmwrapsqueeze{0.3}  % default is 0.4 (40%)
    35→%
    36→%   % Custom squeeze-fit minimum space:
    37→%   \swarmwrapsqueezemin{3\baselineskip}  % default is 2\baselineskip
    38→%
    39→% Zero arguments. Width and height are auto-detected from the rendered
    40→% content box. The user controls the content width explicitly (e.g.,
    41→% \includegraphics[width=3cm] or \begin{minipage}[t]{3cm}).
    42→% Right-side wrapping only. Left-wrap removed in v3.0 (scope reduction).
    43→%
    44→% Requirements: LuaLaTeX (errors out on pdfLaTeX/XeLaTeX)
    45→%
    46→% ── Known Limitations ─────────────────────────────────────────────────
    47→%
    48→% 1. Ghost narrowing (KNOWN LIMITATION, mitigated in v3.33/v3.35):
    49→%    When the wrapped paragraph spans a page break, TeX's parshape
    50→%    persists across the page boundary but the figure does not (it is
    51→%    placed via \smash{\rlap{...}} which is page-local). Continuation
    52→%    lines on the next page remain narrowed even though no figure is
    53→%    beside them.
    54→%    v3.18 attempted to fix this via page-eject but CAUSED A REGRESSION
    55→%    (ghost 4 to 11 on 50-page test). The page-eject leaked parshape
    56→%    state across \newpage. v3.19 REVERTS the page-eject. Instead, the
    57→%    post_linebreak_filter detects page breaks within the narrow zone
    58→%    (total_lines < nl) and zeros out remaining@nl to prevent everypar
    59→%    from amplifying the leak into subsequent paragraphs.
    60→%    v3.33 adds a "penalty fence": penalties inserted between EVERY pair
    61→%    of consecutive narrow lines (not just at the boundary). This forces
    62→%    TeX to keep the entire narrow zone on one page or push it to the
    63→%    next, reducing page-break ghost narrowing. NOTE: v3.35 discovered
    64→%    that the penalty fence was dead code (Lua -- comments ate the
    65→%    callback). v3.35 fixes this, ACTIVATING the penalty fence for
    66→%    the first time.
    67→%    v3.34 adds pre_shipout_filter ghost fix (code correct but callback
    68→%    name misspelled as pre_shipping_filter — no effect until v3.38 fix).
    69→%    v3.35 fixes the critical Lua comment bug, enabling the penalty
    70→%    fence and remaining-nl tracking to actually execute.
    71→%    Remaining instances (inter-figure ghost: narrow text below a figure's
    72→%    bottom extending into the gap before the next figure on the SAME page)
    73→%    are a fundamental TeX limitation (parshape assigned before page
    74→%    breaking).
    75→%
    76→% 2. Deferred figure placement (FIXED in v3.27):
    77→%    When the figure doesn't fit on the current page (\pagegoal - \pagetotal
    78→%    < figure height), v3.27 now forces a new page and places the figure
    79→%    at the top-right with narrow parshape for the following text.
    80→%    Previous approach (\afterpage overlay) was fundamentally broken:
    81→%    it placed a zero-height figure at the top of the next page, but
    82→%    subsequent paragraphs flowed full-width through the overlay.
    83→%    Three independent attempts (v3.24, v3.25, v3.26) confirmed \afterpage
    84→%    cannot work for this use case. v3.27's \newpage approach satisfies
    85→%    wrapping spec #4 ("near newpage: right-wrap on next page").

    89→% v3.35: CRITICAL BUG FIX — Lua callbacks were never executing.
    90→%   ROOT CAUSE: Lua -- comments inside \directlua{...} blocks are
    91→%   fatal. TeX's token scanner converts newlines to spaces before
    92→%   passing the string to Lua. Lua sees `-- comment rest_of_code`
    93→%   as one line, eating everything after the first -- comment.
    94→%   This meant the ENTIRE post_linebreak_filter callback (lines 487-563)
    95→%   was dead code from v3.17 onward. The function was never defined
    96→%   (swarmwrap_post_lb = nil), so the callback was never registered.
    97→%   All features attributed to this callback were actually working via
    98→%   TeX-level mechanisms (everypar, parshape, list patch) alone:
    99→%     - v3.17 remaining-nl tracking: was NOT decrementing (everypar used
   100→%       raw nl, not remaining@nl — but worked because everypar checked
   101→%       remaining@nl > 0 which was only set once at \swarmwrapnext and
   102→%       never decremented, so it stayed positive for the entire wrap zone)
   103→%     - v3.19 cross-page fix: was NOT zeroing remaining@nl on page break
   104→%       (relied on post_linebreak_filter which was dead)

   106→%     - v3.34 pre_shipout_filter: also had -- comments + wrong name, never registered

   108→%       subsequent paragraphs)

   125→%   would leak to the next page via everypar, causing narrow text with no
   126→%   figure beside it. v3.34 zeros remaining@nl and clears everypar at page

   134→% v3.33: Reduce ghost narrowing via penalty fence (Known Limitation #1).
   135→%   Ghost narrowing occurs when a wrapped paragraph spans a page break —
   136→%   TeX assigns parshape (narrow lines) before page breaking, so narrow
   137→%   lines on the continuation page have no figure beside them.
   138→%   FIX: In post_linebreak_filter, insert \penalty between EVERY pair of
   139→%   consecutive narrow lines (not just after the last one). This creates
   140→%   a "penalty fence" that prevents TeX from breaking within the narrow
   141→%   zone. TeX must either keep all narrow lines on the current page (with
   142→%   the figure) or push them all to the next page. This eliminates
   143→%   page-break ghost narrowing without the regressions of v3.18's approach
   144→%   (which used \newpage and leaked parshape state). The penalty value
   145→%   is the user-configurable \swarmwrappenalty (default 10000).

   625→% ── Lua callback: post_linebreak_filter ───────────────────────────────
   626→% This callback:
   627→%   (a) v3.33 penalty fence: inserts penalties between consecutive narrow
   628→%       lines to prevent page breaks within the narrow zone.
   629→%   (b) v3.17 remaining-nl tracking: decrements remaining@nl counter.
   630→%   (c) v3.19 cross-page fix: zeros remaining@nl if paragraph broke
   631→%       across pages (total_lines < nl).
   632→%   (d) v3.29 everypar clearing: clears everypar when remaining@nl = 0.
   633→%   (e) v3.14 remaining height: computes vspace for next \swarmwrapnext.
   634→%   (f) v3.15 emergencystretch reset: prevents leak to subsequent paragraphs.
   635→%   (g) v3.6 boundary penalty: inserts penalty after last narrow line.
   636→% IMPORTANT: Lua -- comments are FORBIDDEN inside \directlua{} blocks.
   637→%   TeX strips newlines, causing -- to consume all subsequent code.
   638→\ifswarmwrap@luatex
   639→\directlua{
   640→  function swarmwrap_post_lb(head, groupcode)
   641→    local tw_sp = tex.dimen["swarmwrap@tw@lua"]
   642→    if tw_sp <= 0 then
   643→      return head
   644→    end
   645→    local tw_val = tw_sp / 65536.0
   646→    local tol = 3.0
   647→    local narrow_width_max = tw_val + tol
   648→    local last_narrow = nil
   649→    local prev_narrow = nil
   650→    local total_lines = 0
   651→    local narrow_count = 0
   652→    local penalty_val = tex.count["swarmwrap@penalty"]
   653→    local current = head
   654→    local narrow_nodes = {}
   655→    while current do
   656→      if current.id == 0 then
   657→        total_lines = total_lines + 1
   658→        local lw = current.width / 65536.0
   659→        if lw <= narrow_width_max and lw > 0 then
   660→          narrow_count = narrow_count + 1
   661→          last_narrow = current
   662→          narrow_nodes[rawlen(narrow_nodes) + 1] = current
   663→        end
   664→      end
   665→      current = current.next
   666→    end
   667→    if penalty_val > 0 and rawlen(narrow_nodes) > 1 then
   668→      for i = 1, rawlen(narrow_nodes) - 1 do
   669→        local p = node.new(node.id("penalty"))
   670→        p.penalty = penalty_val
   671→        node.insert_after(head, narrow_nodes[i], p)
   672→      end
   673→    end
   674→    if tex.count["swarmwrap@remaining@nl"] > 0 then
   675→      local r = tex.count["swarmwrap@remaining@nl"]
   676→      local nl = tex.count["swarmwrap@nl"]
   677→      if total_lines < nl then
   678→        tex.count["swarmwrap@remaining@nl"] = 0
   679→      else
   680→        r = math.max(0, r - narrow_count)
   681→        tex.count["swarmwrap@remaining@nl"] = r
   682→      end
   683→      if tex.count["swarmwrap@remaining@nl"] == 0 then
   684→        local stack = swarmwrap_fig_stack
   685→        if type(stack) == "table" and rawlen(stack) > 0 then
   686→          local prev = stack[rawlen(stack)]
   687→          stack[rawlen(stack)] = nil
   688→          tex.count["swarmwrap@remaining@nl"] = prev.remaining_nl
   689→          if rawlen(stack) > 0 then
   690→            local min_tw = prev.tw
   691→            for i = 1, rawlen(stack) do
   692→              if stack[i].tw < min_tw then
   693→                min_tw = stack[i].tw
   694→              end
   695→            end
   696→            tex.dimen["swarmwrap@tw@lua"] = min_tw
   697→          else
   698→            tex.dimen["swarmwrap@tw@lua"] = prev.tw
   699→          end
   700→        else
   701→          tex.toks["everypar"] = ""
   702→        end
   703→      end
   704→    end
   705→    local nl = tex.count["swarmwrap@nl"]
   706→    local bs = tex.skip["baselineskip"].width
   707→    local excess = 0
   708→    if total_lines < nl then
   709→      excess = (nl - total_lines) * bs
   710→    end
   711→    tex.dimen["swarmwrap@remaining"] = excess
   712→    tex.dimen["emergencystretch"] = 0
   713→    tex.count["interlinepenalty"] = 0
   714→    if penalty_val > 0 and last_narrow then
   715→      local p = node.new(node.id("penalty"))
   716→      p.penalty = penalty_val
   717→      node.insert_after(head, last_narrow, p)
   718→    end
   719→    return head
   720→  end
   721→  luatexbase.add_to_callback("post_linebreak_filter",
   722→    swarmwrap_post_lb, "swarmwrap: penalty at parshape boundary")
   723→}
   724→\fi

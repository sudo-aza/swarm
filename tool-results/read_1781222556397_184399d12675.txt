     1→% swarmwrap.sty — Right-side float wrapper for the Swarm toolkit (v3.47)
     2→%
     3→% Text wrapping using \parshape applied by \swarmwrapnext command.
     4→% Built from scratch after all 9 existing LaTeX wrapping packages failed QA.
     5→%
     6→% Usage:
     7→%   \usepackage{swarmwrap}
     8→%
     9→%   % Simple figure (single content box):
    10→%   \begin{swarmwrap}%
    11→%     \includegraphics[width=3cm]{photo}%
    12→%   \end{swarmwrap}%
    13→%   \swarmwrapnext
    14→%   Paragraph text wraps around the figure.
    15→%
    16→%   % Multi-line content (figure + caption):
    17→%   \begin{swarmwrap}%
    18→%     \begin{minipage}[t]{3cm}
    19→%       \centering
    20→%       \includegraphics[width=\linewidth]{photo}
    21→%       \captionof{figure}{Caption.}
    22→%     \end{minipage}%
    23→%   \end{swarmwrap}%
    24→%   \swarmwrapnext
    25→%   Paragraph text wraps around the figure.
    26→%
    27→%   % Custom page-break penalty (discourage breaks in wrapped zone):
    28→%   \swarmwrappenalty{500}  % default is 10000 (very discouraged)
    29→%
    30→%   % Custom text-figure gap (horizontal space between text and figure):
    31→%   \swarmwrapgap{20pt}  % default is 14pt
    32→%
    33→%   % Custom squeeze-fit threshold (fraction of figure height):
    34→%   \swarmwrapsqueeze{0.3}  % default is 0.4 (40%)
    35→%
    36→%   % Custom squeeze-fit minimum space:
    37→%   \swarmwrapsqueezemin{3\baselineskip}  % default is 2\baselineskip
    38→%
    39→% Zero arguments. Width and height are auto-detected from the rendered
    40→% content box. The user controls the content width explicitly (e.g.,
    41→% \includegraphics[width=3cm] or \begin{minipage}[t]{3cm}).
    42→% Right-side wrapping only. Left-wrap removed in v3.0 (scope reduction).
    43→%
    44→% Requirements: LuaLaTeX (errors out on pdfLaTeX/XeLaTeX)
    45→%
    46→% ── Known Limitations ─────────────────────────────────────────────────
    47→%
    48→% 1. Ghost narrowing (FIXED in v3.47):
    49→%    When the wrapped paragraph spans a page break, TeX's parshape
    50→%    persists across the page boundary but the figure does not (it is
    51→%    placed via \smash{\rlap{...}} which is page-local). Continuation
    52→%    lines on the next page remain narrowed even though no figure is
    53→%    beside them.
    54→%    v3.18 attempted to fix this via page-eject but CAUSED A REGRESSION
    55→%    (ghost 4 to 11 on 50-page test). The page-eject leaked parshape
    56→%    state across \newpage. v3.19 REVERTS the page-eject. Instead, the
    57→%    post_linebreak_filter detects page breaks within the narrow zone
    58→%    (total_lines < nl) and zeros out remaining@nl to prevent everypar
    59→%    from amplifying the leak into subsequent paragraphs.
    60→%    v3.33 adds a "penalty fence": penalties inserted between EVERY pair
    61→%    of consecutive narrow lines (not just at the boundary). This forces
    62→%    TeX to keep the entire narrow zone on one page or push it to the
    63→%    next, reducing page-break ghost narrowing. NOTE: v3.35 discovered
    64→%    that the penalty fence was dead code (Lua -- comments ate the
    65→%    callback). v3.35 fixes this, ACTIVATING the penalty fence for
    66→%    the first time.
    67→%    v3.34 adds pre_shipping_filter ghost fix (code present but callback
    68→%    fails to register due to format cache — no effect).
    69→%    v3.35 fixes the critical Lua comment bug, enabling the penalty
    70→%    fence and remaining-nl tracking to actually execute.
    71→%    v3.47 fixes remaining continuation ghost narrowing by widening
    72→%    orphaned narrow hboxes in post_linebreak_filter. When the first
    73→%    page's portion of a broken paragraph has total_lines < nl, a Lua
    74→%    flag (swarmwrap_ghost_widen) is set. The next post_linebreak_filter
    75→%    invocation (continuation page) detects this flag and widens all
    76→%    narrow hboxes to \linewidth by appending \hfil, restoring full-width
    77→%    text on the continuation page where no figure exists.
    78→%
    79→% 2. Multi-figure stacking (FIXED in v3.37):
    80→%    When multiple figures are stacked on the same page, earlier tall
    81→%    figures may still be rendering after a later figure's narrow lines
    82→%    are exhausted. v3.37 maintains a Lua stack of active figures
    83→%    (swarmwrap_fig_stack). Each \swarmwrapnext pushes an entry; the
    84→%    post_linebreak_filter tracks remaining lines per figure; when the
    85→%    top figure is exhausted, the previous figure's parameters are
    86→%    promoted. Text only returns to full width when all figures on the
    87→%    page are exhausted.
    88→%
    89→% 3. Deferred figure placement (FIXED in v3.27):
    90→%    When the figure doesn't fit on the current page (\pagegoal - \pagetotal
    91→%    < figure height), v3.27 now forces a new page and places the figure
    92→%    at the top-right with narrow parshape for the following text.
    93→%    Previous approach (\afterpage overlay) was fundamentally broken:
    94→%    it placed a zero-height figure at the top of the next page, but
    95→%    subsequent paragraphs flowed full-width through the overlay.
    96→%    Three independent attempts (v3.24, v3.25, v3.26) confirmed \afterpage
    97→%    cannot work for this use case. v3.27's \newpage approach satisfies
    98→%    wrapping spec #4 ("near newpage: right-wrap on next page").
    99→%

   102→% v3.47: Fix continuation ghost narrowing (user challenge — "unacceptable").
   103→%   BUG: When a wrapped paragraph is long enough to span a page break,
   104→%   TeX's parshape persists across the page boundary but the figure
   105→%   (\smash{\rlap}) does not. Continuation lines on the next page are
   106→%   narrowed with no figure beside them. The penalty fence (v3.33)
   107→%   prevents MOST breaks but cannot stop all — TeX always commits at
   108→%   least 1 line before seeing a penalty, and very long paragraphs with
   109→%   tall figures can still break within the narrow zone despite the fence.
   110→%   Previous attempts to fix this:
   111→%     v3.18: Page-eject → REGRESSED (4→11 ghost, parshape leak)
   112→%     v3.19: Reverted v3.18
   113→%     v3.34: pre_shipping_filter state reset → dead code (callback
   114→%            never registered, later fixed in v3.35 but only handles
   115→%            multi-paragraph ghost, not single-paragraph continuation)
   116→%   ROOT CAUSE: TeX assigns parshape before page breaking. The paragraph
   117→%   breaker produces hboxes at the narrow width. When the page breaks
   118→%   mid-paragraph, LuaTeX calls post_linebreak_filter twice — once per
   119→%   page portion. The second call receives the continuation lines that
   120→%   are already set to narrow width. No TeX-level mechanism can prevent
   121→%   this because parshape is consumed during line breaking, which
   122→%   precedes page breaking.
   123→%   FIX: In post_linebreak_filter, when total_lines < nl (paragraph broke
   124→%   across pages within narrow zone), set a Lua flag swarmwrap_ghost_widen
   125→%   with the original narrow width value. On the NEXT invocation of
   126→%   post_linebreak_filter (processing the continuation page's lines),
   127→%   detect the flag and widen all narrow hboxes back to \linewidth by:
   128→%   (1) Setting each hbox's width to \linewidth
   129→%   (2) Appending an \hfil glue node at the end of each hbox
   130→%   This restores full-width text on the continuation page. The text
   131→%   content and line breaks are preserved — only the box width and
   132→%   right margin glue change. After widening, clear the flag.
   133→%   TRADE-OFF: The widened lines may have slightly different line endings
   134→%   than if they had been originally set at full width (the line breaks
   135→%   were computed for the narrow width). However, since the text is
   136→%   left-aligned and \hfil fills the right margin, the visual result
   137→%   is clean full-width text — far better than narrow ghost lines.

   806→%       lines to prevent page breaks within the narrow zone.
   807→%   (b) v3.17 remaining-nl tracking: decrements remaining@nl counter.

   809→%       across pages (total_lines < nl).

   812→%   (f) v3.15 emergencystretch reset: prevents leak to subsequent paragraphs.
   813→%   (g) v3.6 boundary penalty: inserts penalty after last narrow line.
   814→%   (h) v3.37 figure stack: pops exhausted figures and promotes next
   815→%       entry so everypar continues with previous figure's parameters.

   821→    if swarmwrap_ghost_widen then
   822→      local lw_val = swarmwrap_safe_dimen("linewidth")
   823→      local ghost_tw = swarmwrap_ghost_tw
   824→      local tol = 3.0
   825→      local narrow_max = (ghost_tw / 65536.0) + tol
   826→      local current = head

   828→        if current.id == 0 then
   829→          local lw = current.width / 65536.0
   830→          if lw <= narrow_max and lw > 0 then
   831→            current.width = lw_val
   832→            local fil = node.new(node.id("glue"))
   833→            fil.stretch = 65536
   834→            fil.stretch_order = 2
   835→            node.insert_after(head, current, fil)
   836→          end
   837→        end
   838→        current = current.next
   839→      end
   840→      swarmwrap_ghost_widen = false
   841→      swarmwrap_ghost_tw = 0
   842→      return head
   843→    end
   844→    local tw_sp = swarmwrap_safe_dimen("swarmwrap@tw@lua")
   845→    if tw_sp <= 0 then
   846→      return head
   847→    end
   848→    local tw_val = tw_sp / 65536.0
   849→    local tol = 3.0
   850→    local narrow_width_max = tw_val + tol
   851→    local last_narrow = nil
   852→    local prev_narrow = nil
   853→    local total_lines = 0
   854→    local narrow_count = 0
   855→    local penalty_val = swarmwrap_safe_count("swarmwrap@penalty")
   856→    local current = head
   857→    local narrow_nodes = {}
   858→    while current do
   859→      if current.id == 0 then
   860→        total_lines = total_lines + 1
   861→        local lw = current.width / 65536.0
   862→        if lw <= narrow_width_max and lw > 0 then
   863→          narrow_count = narrow_count + 1
   864→          last_narrow = current
   865→          narrow_nodes[#narrow_nodes + 1] = current
   866→        end
   867→      end
   868→      current = current.next
   869→    end
   870→    if penalty_val > 0 and #narrow_nodes > 1 then
   871→      for i = 1, #narrow_nodes - 1 do
   872→        local p = node.new(node.id("penalty"))
   873→        p.penalty = penalty_val
   874→        node.insert_after(head, narrow_nodes[i], p)
   875→      end
   876→    end
   877→    if swarmwrap_safe_count("swarmwrap@remaining@nl") > 0 then
   878→      local r = swarmwrap_safe_count("swarmwrap@remaining@nl")
   879→      local nl = swarmwrap_safe_count("swarmwrap@nl")
   880→      if total_lines < nl then
   881→        swarmwrap_ghost_widen = true
   882→        swarmwrap_ghost_tw = tw_sp
   883→        tex.count["swarmwrap@remaining@nl"] = 0
   884→        swarmwrap_fig_stack = {}
   885→        tex.toks["everypar"] = {}
   886→      else
   887→        r = math.max(0, r - narrow_count)
   888→        tex.count["swarmwrap@remaining@nl"] = r
   889→        local stack = swarmwrap_fig_stack
   890→        if type(stack) ~= "table" then
   891→          stack = {}
   892→          swarmwrap_fig_stack = stack
   893→        end
   894→        if #stack > 0 then
   895→          stack[#stack].remaining_nl = r
   896→        end
   897→      end
   898→      if swarmwrap_safe_count("swarmwrap@remaining@nl") == 0 then
   899→        local stack = swarmwrap_fig_stack
   900→        if #stack > 1 then
   901→          table.remove(stack)
   902→          local prev = stack[#stack]
   903→          tex.count["swarmwrap@remaining@nl"] = prev.remaining_nl
   904→          tex.dimen["swarmwrap@tw@lua"] = prev.tw
   905→          tex.count["swarmwrap@nl"] = prev.nl
   906→          local min_tw = prev.tw
   907→          for i = 1, #stack do
   908→            if stack[i].tw > 0 and stack[i].tw < min_tw then
   909→              min_tw = stack[i].tw
   910→            end
   911→          end
   912→          if min_tw <= 0 then
   913→            min_tw = prev.tw
   914→          end
   915→          tex.dimen["swarmwrap@tw@lua"] = min_tw
   916→        else
   917→          if #stack == 1 then
   918→            table.remove(stack)
   919→          end
   920→          tex.toks["everypar"] = {}
   921→        end
   922→      end
   923→    end
   924→    local nl = swarmwrap_safe_count("swarmwrap@nl")
   925→    local bs = swarmwrap_safe_dimen("baselineskip")
   926→    local excess = 0
   927→    if total_lines < nl then
   928→      excess = (nl - total_lines) * bs
   929→    end
   930→    tex.dimen["swarmwrap@remaining"] = excess
   931→    tex.dimen["emergencystretch"] = 0
   932→    if penalty_val > 0 and last_narrow then
   933→      local p = node.new(node.id("penalty"))
   934→      p.penalty = penalty_val
   935→      node.insert_after(head, last_narrow, p)
   936→    end
   937→    if penalty_val > 0 and swarmwrap_safe_count("swarmwrap@remaining@nl") > 0 then
   938→      local last_node = head
   939→      while last_node.next do
   940→        last_node = last_node.next
   941→      end
   942→      local p = node.new(node.id("penalty"))
   943→      p.penalty = penalty_val
   944→      node.insert_after(head, last_node, p)
   945→    end
   946→    return head
   947→  end
   948→  luatexbase.add_to_callback("post_linebreak_filter",
   949→    swarmwrap_post_lb, "swarmwrap: penalty at parshape boundary")
   950→}
   951→\fi

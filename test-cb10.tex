\documentclass{article}
\usepackage{swarmwrap}
\directlua{
  local list = callback.list()
  if list["post_linebreak_filter"] then
    texio.write_nl("RAW CALLBACK EXISTS: " .. tostring(list["post_linebreak_filter"]))
  else
    texio.write_nl("RAW CALLBACK NOT FOUND")
  end
  -- Also try luatexbase
  local desc = luatexbase.callback_descriptions("post_linebreak_filter")
  texio.write_nl("LUATEXBASE DESC TYPE: " .. type(desc))
  if type(desc) == "string" then
    texio.write_nl("DESC: " .. desc)
  elseif type(desc) == "table" then
    for k, v in ipairs(desc) do
      texio.write_nl("  CB " .. k .. ": " .. tostring(v))
    end
  end
}

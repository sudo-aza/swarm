Test page.
\directlua{
  local f = io.open("/tmp/bp-test-output.txt", "w")
  -- Test tex.dimen access to pagegoal
  local ok1, err1 = pcall(function()
    local v = tex.dimen["pagegoal"]
    f:write("dimen pagegoal = " .. tostring(v) .. "\n")
  end)
  if not ok1 then f:write("dimen FAIL: " .. tostring(err1) .. "\n") end

  -- Test direct tex.pagegoal
  local ok2, err2 = pcall(function()
    local v = tex.pagegoal
    f:write("tex.pagegoal = " .. tostring(v) .. "\n")
  end)
  if not ok2 then f:write("pagegoal FAIL: " .. tostring(err2) .. "\n") end

  -- Test pagetotal
  local ok3, err3 = pcall(function()
    local v = tex.dimen["pagetotal"]
    f:write("dimen pagetotal = " .. tostring(v) .. "\n")
  end)
  if not ok3 then f:write("pagetotal FAIL: " .. tostring(err3) .. "\n") end

  f:close()
}
\end{document}

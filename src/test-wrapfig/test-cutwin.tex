% cutwin API: \begin{cutout}{numtop}{leftwidth}{rightwidth}{numcut}
%   numtop     = full-width lines above window
%   numcut     = number of cutout lines (window height)
%   leftwidth  = text line width LEFT of window (ignored for open-right)
%   rightwidth = text line width RIGHT of window (ignored for open-left)
%   For centered: both are used (left and right text widths)
% Body of env = text paragraph (must be single paragraph)
% Window content via \renewcommand{\windowpagestuff}{...}

% Test 4: Itemize inside cutout
% FAIL — itemize does not respect parshape constraint.
% Third item overflows cutout boundary by 49pt (w=290 vs ~241pt).
% Overfull hbox (43.8pt) confirms the violation. Items 1-2 wrap
% correctly at ~246pt (incl bullet indent), but item 3+ extend into
% the figure zone. This is a fundamental limitation: itemize uses
% its own list machinery that does not cooperate with parshape.
% ============================================================
\section{Test 4: Itemize inside cutout}

\opencutright
\renewcommand{\windowpagestuff}{%
  \centering\vspace*{0.5em}%
  \includegraphics[width=3cm]{test-image.png}%
}
\begin{cutout}{1}{8.5cm}{0pt}{10}
  Here is a list of items about this figure. Each item provides some
  relevant detail.
  \begin{itemize}
    \item First item in the list, should wrap around the figure.
    \item Second item, also wrapping.
    \item Third item, may or may not wrap depending on figure height.
    \item Fourth item.
    \item Fifth item, likely at full width.
  \end{itemize}
  This is trailing text after the list.
\end{cutout}

% test-figflow.tex — figflow (plain TeX) test
% figflow API: \figflow{width}{height}{content}
%   width  = figure width (negative = right, positive = left)
%   height = total figure height including caption
%   content = figure material + optional \vfill caption text
%
% figflow auto-ends when enough text has wrapped (\prevgraf > \figlines).
% Do NOT call \endflow manually — it corrupts \par (double \endgroup).
% The source says "Does not work for Latex" because \line conflicts with
% LaTeX's picture mode. Workaround: redefine \line to \hbox to\hsize.
%
% figflow uses \hangindent/\hangafter (not parshape) and redefines \par.
% It checks \pageremains — if the figure won't fit, it moves to the next page.
%
% CRITICAL: Must use \input{figflow} (plain TeX macro, not a LaTeX package).
\documentclass[11pt,a4paper]{article}
\usepackage[utf8]{inputenc}
\usepackage{graphicx}
\usepackage{lipsum}
\usepackage{multicol}

% figflow is a plain TeX macro — load and fix \line conflict
\makeatletter
\input{figflow}
\let\LaTeXline\line
\def\line{\hbox to\hsize}
\makeatother

\title{figflow (plain TeX) Test}
\author{Programmer Agent}
\date{2026-05-16}

\begin{document}
\maketitle

% ============================================================
% Test 1: Basic right-side figure (negative width)
% RESULT: PASS. Image at (386,320)-(471,384) right. Text wraps at
% w=242-259pt (full width 359pt). Caption at (384,416). "Moving figure..."
% message printed — figflow deferred placement until enough room.
% ============================================================
\section{Test 1: Basic right-side figure}

\figflow{-3.5cm}{4cm}{%
  \vbox to 3.5cm{\vfil\centering\includegraphics[width=3cm]{test-image.png}\vfil}%
  \vfill\noindent Figure 1: Right side.%
}
\lipsum[1-2]

% ============================================================
\section{Test 2: Basic left-side figure}

\figflow{3.5cm}{4cm}{%

% ============================================================

% page 2 ("Moving figure...") and moved 8cm figure to page 3.

% Unlike parshape packages, figflow gracefully defers — no overflow.
% ============================================================
\section{Test 3: Tall figure near page break}

\lipsum[4-5]

\figflow{-3.5cm}{8cm}{%
  \vbox to 7cm{\vfil\centering\includegraphics[width=3cm,height=7cm,keepaspectratio=false]{test-image.png}\vfil}%
  \vfill\noindent Figure 3: Tall figure.%
}
This text should flow beside the figure. Since figflow checks available
space via \texttt{\textbackslash pageremains}, the figure should be placed
on whichever page has enough room.
\lipsum[6-7]

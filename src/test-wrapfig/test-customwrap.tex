% Test 7: Page Break Detection (v2.2 feature)
% Expected: PASS. A vbox fills exactly (textheight - 80pt), leaving
%   ~80pt remaining. The figure needs ~137pt (3.8cm rule + caption +
%   baselineskip). Since 80pt < 137pt, \swarmwrapnext should insert
%   \newpage, pushing the wrapped block to the next page. No overfull
%   vbox warnings should appear on ANY page.
% ====================================================================
\section*{Test 7: Page Break Detection}
This test verifies that swarmwrap.sty v2.2 detects insufficient
remaining page space and inserts \texttt{\textbackslash newpage}
before the wrapped block.

% Fill exactly enough space to leave ~80pt remaining on this page.
\newdimen\swtestsevenht
\swtestsevenht=\textheight
\advance\swtestsevenht by -80pt

\vbox to\swtestsevenht{%
  This paragraph and the following text fill most of the page,
  leaving approximately 80pt of remaining space. The swarmwrap
  figure requires about 137pt (3.8cm rule + caption +
  baselineskip), which exceeds the remaining space. Therefore,
  the page break detection in \texttt{\textbackslash swarmwrapnext}
  should insert \texttt{\textbackslash newpage} before the
  wrapped block, pushing it cleanly to the next page.%
  \lipsum[1-3]%
  \vss
}%

  \rule{\linewidth}{3.8cm}
  \captionof{figure}{Near-bottom figure (3cm wide, 4cm tall).}
  \label{fig:near-bottom}
\end{swarmwrap}

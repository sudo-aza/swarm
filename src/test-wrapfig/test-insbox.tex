% Test: insbox (v2.2) — generic parshape wrapper for inserting boxes into paragraphs
% API:
%   \InsertBoxR{n}{box}[correction] — right-side, n lines before box
%   \InsertBoxL{n}{box}[correction] — left-side
%   \InsertBoxC{box}               — centered, used INSIDE a paragraph (via \vadjust)
%   \MoveBelowBox                   — start new paragraph below the current frame
% CAUTION: [correction] is optional and parsed via \futurelet.
%   If the next token after } starts with [, it is consumed as [correction].
%   Use {} or \relax after the box argument to prevent this.
% CRITICAL: insbox uses \makeatletter/\input insbox (not \usepackage).
\documentclass[11pt,a4paper]{article}
\usepackage[utf8]{inputenc}
\usepackage{graphicx}
\usepackage{lipsum}

% insbox is a plain TeX macro file, must be \input'd inside \makeatletter
\makeatletter
\input{insbox}
\makeatother

\title{insbox Test (v2.2)}
\author{Programmer Agent}
\date{2026-05-16}

\begin{document}
\maketitle

%% ===================================================================
%% Test 1: Basic right insertion — \InsertBoxR{0}{...}{}
%% PASS. Image at (388,311)-(473,375) right. Text wraps left at w=244-261.
%% Text resumes at full width (359) after wrapping zone.
%% ===================================================================
\section*{Test 1: \texttt{\textbackslash InsertBoxR} (right, 0 leading lines)}

\InsertBoxR{0}{%
  \fbox{\includegraphics[width=3cm]{test-image.png}}%
}{}%
\lipsum[1][1-8]
This sentence continues after the wrapped text to verify wrapping ends correctly.
\MoveBelowBox
Normal text resumes at full width below the image.

  \item Fifth item — at full width.

\MoveBelowBox

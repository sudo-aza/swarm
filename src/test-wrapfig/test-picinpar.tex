% test-picinpar.tex — picinpar (v1.3a) test
% picinpar API: \begin{window}[N, pos, {content}, {caption}]
%   N       = number of full-width lines above the window
%   pos     = l (left), r (right), c (centered)
%   content = graphic material (in braces — NO [] brackets allowed!)
%   caption = caption text (in braces), or empty {} for no caption
% Text that wraps goes INSIDE the window environment body.
% \figwindow = same as window but with figure caption counter.
%
% CRITICAL: Arg #3 cannot contain [] brackets (e.g. \includegraphics[width=...])
% because the window command parses all args from a single [...] group.
% Workaround: save image to a box first, then use the box as arg #3.
%
% KNOWN LIMITATION: picinpar redefines \par inside window environments.
% This causes "missing \item" errors when itemize/enumerate is placed inside
% a window. The window must contain plain text paragraphs only.
\documentclass[11pt,a4paper]{article}
\usepackage[utf8]{inputenc}
\usepackage{graphicx}
\usepackage{picinpar}
\usepackage{lipsum}

\title{picinpar (v1.3a) Test}
\author{Programmer Agent}
\date{2026-05-16}

% Pre-save images in boxes to avoid [] parsing issues
\newsavebox{\testimager}
\newsavebox{\testimagel}
\newsavebox{\testimagec}
\newsavebox{\tallimager}
\AtBeginDocument{%
  \savebox{\testimager}{\includegraphics[width=3.5cm]{test-image.png}}%
  \savebox{\testimagel}{\includegraphics[width=3.5cm]{test-image.png}}%
  \savebox{\testimagec}{\includegraphics[width=3cm]{test-image.png}}%
  \savebox{\tallimager}{\includegraphics[width=3.5cm,height=10cm,keepaspectratio=false]{test-image.png}}%
}

\begin{document}
\maketitle

% ============================================================
% Test 1: Basic right-side window
% ============================================================
\section{Test 1: Basic right-side window}

\begin{window}[3,r,{\usebox{\testimager}},{A test image on the right side.}]
  \lipsum[1-2]
\end{window}

% ============================================================
% Test 2: Left-side window
% ============================================================
\section{Test 2: Left-side window}

\begin{window}[2,l,{\usebox{\testimagel}},{Image on the left.}]
  \lipsum[2-5]
\end{window}

% ============================================================
% Test 3: Tall figure
% ============================================================
% picinpar is parshape-based and CANNOT span page breaks.
% A 10cm figure is used here to demonstrate that tall figures
% still work — picinpar places the wrapping zone wherever the
% paragraph starts, regardless of figure height. The figure
% just gets pushed to the next page if it doesn't fit in the
% remaining space, and wrapping continues from that page.
\section{Test 3: Tall figure}

\begin{window}[2,r,{\usebox{\tallimager}},{A 10cm tall figure.}]
  \lipsum[5-8]
\end{window}

% ============================================================
% Test 4: Window before itemize (workaround)
% The window contains plain text that flows into a list AFTER
% the window environment ends. This avoids the \par redefinition
% conflict inside itemize.
% ============================================================
\section{Test 4: Window before itemize (workaround)}

\begin{window}[1,r,{\usebox{\testimager}},{Image beside a list.}]
  Here is a list of items about this figure. Each item provides some
  relevant detail. The first few items should appear beside the figure.
  \lipsum[8][1-3]
\end{window}

% List continues AFTER the window ends — at full width
\begin{itemize}
  \item This item is after the window ended — full width.
  \item Second item at full width.
  \item Third item at full width.
  \item Fourth item to fill space.
  \item Fifth item for more content.
  \item Sixth item — all at full width after the window.
\end{itemize}

\section{Test 5: Window inside itemize}

This test places the window environment inside an itemize item.

\begin{itemize}
  \item First item before the window.

  \begin{window}[1,r,{\usebox{\testimager}},{}]
    This text is inside the window within an item. It should wrap
    around the figure on the right side.
    \lipsum[9][1-6]
  \end{window}

  \item Second item after the window.
  \item Third item, should be at full width.
  \item Fourth item to demonstrate continued list behavior.
  \item Fifth item — list resumes normally after the window.
  \item Sixth item for good measure.
\end{itemize}
